# BEGIN tile 0

1         # (int) tile_id
forest    # (char *) sprite

-10.5     # (float) resource 0 multiplier

177261    # (int) quantity of resource 0 available
12001     # (int) quantity of resource 1 available

120       # number of enemies

# END TILE 0
